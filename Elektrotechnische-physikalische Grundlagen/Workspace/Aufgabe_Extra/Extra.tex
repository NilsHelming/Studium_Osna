	&& \ln\left(\frac{MTTF_1}{MTTF_2}\right) &= \frac{W_a}{k \cdot T_1}-\frac{W_a}{k \cdot T_2}&&\\
	&& k \cdot \ln\left(\frac{MTTF_1}{MTTF_2}\right) &= W_a\left(\frac{1}{T_1}-\frac{1}{T_2}\right)&&\\
\end{align*}
